Рекурсивные функции

Допишите определение функции pwr от a и b, которая определяется через себя и mpy и выдаёт a ** b для положительных целых чисел.

Открыв консоль браузера, можно вызвать функцию pwr.

          
const sec = a => a + 1;
const add = (a, b) => (b === 0) ? a : sec(add(a, b - 1));
const mpy = (a, b) => (b === 1) ? a : add(a, mpy(a, b - 1));
const pwr = (a, b) => (b === 0) ? 1 : (b === 1) ? a : mpy(a, pwr(a, b - 1));
console.log(pwr(2, 13)); //MAX VAL IS 13
console.log(pwr(2, 14)); //Maximum call stack size exceeded
