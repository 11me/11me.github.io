Цепочка прототипов

Импортируйте (с помощью директивы require или import) объект o3 под именем obj из модуля, код которого находится по адресу: https://kodaktor.ru/j/protochain или https://kodaktor.ru/j/protochain2.

Определите цепочку прототипов этого объекта и экспортируйте результат в качестве массива имён этих прототипа.

Результат можно посмотреть в консоли браузера.

          
import o3 from './goss_proto.js';

function getProtos(obj, arr = []) {

  if (obj !== null) {
    arr.push(obj.name);
    getProtos(Object.getPrototypeOf(obj), arr);
  }
  return arr;

}

const names = getProtos(Object.getPrototypeOf(o3));
// [ 'JavaScript', 'LiveScript', 'Mocha' ]
